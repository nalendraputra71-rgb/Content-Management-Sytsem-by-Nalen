56 const SocialThumbnail = ({ url, fallback }: { url: string, fallback: any }) => {
57   const [img, setImg] = useState<string|null>(null);
58   useEffect(() => {
59     if(!url) return;
60     let isMounted = true;
61     fetch(`https://api.microlink.io/?url=${encodeURIComponent(url)}`)
62       .then(r=>r.json())
63       .then(d=>{
64         if(isMounted && d?.data?.image?.url) {
65           setImg(d.data.image.url);
66         }
67       })
68       .catch(e=>console.log("no thumb"));
69     return () => { isMounted = false };
70   }, [url]);
71 
72   if(img) return <img src={img} alt="thumb" style={{width: "100%", height: "100%", objectFit: "cover"}} />;
73   return fallback;
74 };

77 function CustomDropdown({ value, options = [], onChange, style }: { value: string, options?: any[], onChange: (val: string) => void, style?: any }) {
78   const [open, setOpen] = useState(false);
79   const ref = useRef<HTMLDivElement>(null);
80 
81   useEffect(() => {
82     function handleClickOutside(event: MouseEvent) {
83       if (ref.current && !ref.current.contains(event.target as Node)) setOpen(false);
84     }
85     document.addEventListener("mousedown", handleClickOutside);
86     return () => document.removeEventListener("mousedown", handleClickOutside);
87   }, []);
88 
89   const activeOption = options.find(o => (typeof o === 'string' ? o : o.id) === value);
90   const displayLabel = activeOption ? (typeof activeOption === 'string' ? activeOption : activeOption.label) : value;
91 
92   return (
93     <div ref={ref} style={{ position: "relative", ...style }}>
94       <button 
95         onClick={() => setOpen(!open)} 
96         className="hover-scale"
97         style={{ width: "100%", display: "flex", alignItems: "center", justifyContent: "space-between", gap: 12, padding: "10px 20px", borderRadius: "9999px", border: "1px solid rgba(255,255,255,0.7)", background: "rgba(255,255,255,0.5)", backdropFilter: "none", fontSize: 12, fontWeight: 700, cursor: "pointer", color: "#2C2016" }}
98       >
99         <span>{displayLabel}</span>
100         <ChevronDown size={14} color="rgba(44,32,22,0.5)" style={{ transform: open ? 'rotate(180deg)' : 'none', transition: 'all 0.2s' }} />
101       </button>
102       <AnimatePresence>
103         {open && (
104           <motion.div 
105             initial={{ opacity: 0, y: 5 }} animate={{ opacity: 1, y: 0 }} exit={{ opacity: 0, y: 5 }} transition={{ duration: 0.15 }}
106             style={{ position: "absolute", top: "100%", right: 0, marginTop: 4, background: "rgba(255,255,255,0.85)", backdropFilter: "none", WebkitBackdropFilter: "none", border: "1px solid rgba(255,255,255,0.8)", borderRadius: 12, padding: 6, zIndex: 100, boxShadow: "0 10px 40px rgba(0,0,0,0.1)", minWidth: 120, overflowY: "auto", maxHeight: 200 }}
107           >
108             {options.map((o, i) => {
109               const val = typeof o === 'string' ? o : o.id;
110               const isSelected = val === value;
111               return (
112                 <div 
113                   key={i} 
114                   onClick={() => { onChange(val); setOpen(false); }}
115                   style={{ padding: "8px 12px", borderRadius: 8, fontSize: 12, fontWeight: isSelected?800:600, cursor: "pointer", background: isSelected ? "rgba(59,130,246,0.1)" : "transparent", color: isSelected ? "#3B82F6" : "#2C2016", transition: "all 0.1s", whiteSpace: "nowrap" }}
116                   onMouseEnter={(e) => { if (!isSelected) e.currentTarget.style.background = "rgba(255,255,255,0.6)"; }}
117                   onMouseLeave={(e) => { if (!isSelected) e.currentTarget.style.background = "transparent"; }}
118                 >
119                   {typeof o === 'string' ? o : o.label}
120                 </div>
121               );
122             })}
123           </motion.div>
124         )}
125       </AnimatePresence>
126     </div>
127   );
128 }

187 function DateFilterPopover({ dateFilt, setDateFilt, customS, setCustomS, customE, setCustomE }: any) {
188   const { lang } = useI18n();
189   const [open, setOpen] = useState(false);
190   const ref = useRef<HTMLDivElement>(null);
191 
192   const OPTIONS = [
193     {id:"yesterday", label:"Yesterday"},
194     {id:"7d", label:"Last 7 days"},
195     {id:"28d", label:"Last 28 days"},
196     {id:"90d", label:"Last 90 days"},
197     {id:"tw", label:"This week"},
198     {id:"tm", label:"This month"},
199     {id:"ty", label:"This year"},
200     {id:"lw", label:"Last week"},
201     {id:"lm", label:"Last month"},
202     {id:"custom", label:"Custom"},
203   ];
204 
205   let activeLabel = OPTIONS.find(o => o.id === dateFilt)?.label || (lang === "id" ? "Sepanjang Waktu" : "All Time");
206   if (dateFilt === "custom") {
207     const formatDate = (dStr: string) => {
208       if (!dStr) return "";
209       const d = new Date(dStr);
210       return isNaN(d.getTime()) ? "" : d.toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' });
211     };
212     if (customS && customE) activeLabel = `${formatDate(customS)} - ${formatDate(customE)}`;
213     else if (customS) activeLabel = formatDate(customS);
214   }
215 
216   const [tempFilt, setTempFilt] = useState(dateFilt);
217   const [tempS, setTempS] = useState(customS);
218   const [tempE, setTempE] = useState(customE);
219 
220   useEffect(() => {
221     function handleClickOutside(event: MouseEvent) {
222       if (ref.current && !ref.current.contains(event.target as Node)) {
223         setOpen(false);
224       }
225     }
226     document.addEventListener("mousedown", handleClickOutside);
227     return () => document.removeEventListener("mousedown", handleClickOutside);
228   }, []);
229 
230   const handleUpdate = () => {
231     setDateFilt(tempFilt);
232     if(tempFilt === "custom") {
233       setCustomS(tempS);
234       setCustomE(tempE);
235     }
236     setOpen(false);
237   };
238 
239   const handleCancel = () => {
240     setTempFilt(dateFilt);
241     setTempS(customS);
242     setTempE(customE);
243     setOpen(false);
244   };
245 
246   const [navDate, setNavDate] = useState(new Date());
247 
248   const handleDateClick = (y: number, m: number, d: number) => {
249     setTempFilt("custom");
250     const clickedDateStr = `${y}-${String(m + 1).padStart(2, '0')}-${String(d).padStart(2, '0')}`;
251     
252     if (!tempS || (tempS && tempE)) {
253       setTempS(clickedDateStr);
254       setTempE("");
255     } else {
256       const sDate = new Date(tempS);
257       const clickedDate = new Date(clickedDateStr);
258       if (clickedDate < sDate) {
259         setTempE(tempS);
260         setTempS(clickedDateStr);
261       } else {
262         setTempE(clickedDateStr);
263       }
264     }
265   };
266 
267   const activeDateRange = useMemo(() => {
268     if (tempFilt === "custom" || tempFilt === "all") return { sDateStr: tempS, eDateStr: tempE };
269     const now = new Date();
270     now.setHours(0,0,0,0);
271     let targetS = new Date(now);
272     let targetE = new Date(now);
273     const dayOfWeek = now.getDay();
274 
275     if(tempFilt==="yesterday") {
276       targetS.setDate(now.getDate()-1);
277       targetE = new Date(targetS);
278     } else if(tempFilt==="7d") {
279       targetS.setDate(now.getDate()-7);
280     } else if(tempFilt==="28d") {
281       targetS.setDate(now.getDate()-28);
282     } else if(tempFilt==="90d") {
283       targetS.setDate(now.getDate()-90);
284     } else if(tempFilt==="tw") {
285       targetS.setDate(now.getDate() - dayOfWeek);
286     } else if(tempFilt==="tm") {
287       targetS.setDate(1);
288     } else if(tempFilt==="ty") {
289       targetS.setMonth(0);
290       targetS.setDate(1);
291     } else if(tempFilt==="lw") {
292       targetS.setDate(now.getDate() - dayOfWeek - 7);
293       targetE = new Date(targetS);
294       targetE.setDate(targetS.getDate() + 6);
295     } else if(tempFilt==="lm") {
296       targetS.setMonth(now.getMonth()-1);
297       targetS.setDate(1);
298       targetE = new Date(now.getFullYear(), now.getMonth(), 0); 
299     }
300 
301     const fmt = (d: Date) => {
302       if (isNaN(d.getTime())) return "";
303       return `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}-${String(d.getDate()).padStart(2, '0')}`;
304     };
305     return { sDateStr: fmt(targetS), eDateStr: fmt(targetE) };
306   }, [tempFilt, tempS, tempE]);
307 
308   const isExactDate = (y: number, m: number, d: number) => {
309     if (tempFilt === "all") return false;
310     const { sDateStr, eDateStr } = activeDateRange;
311     if (!sDateStr) return false;
312     const curDateStr = `${y}-${String(m + 1).padStart(2, '0')}-${String(d).padStart(2, '0')}`;
313     return curDateStr === sDateStr || curDateStr === eDateStr;
314   };
315 
316   const isDateInRange = (y: number, m: number, d: number) => {
317     if (tempFilt === "all") return false;
318     const { sDateStr, eDateStr } = activeDateRange;
319     if (!sDateStr || !eDateStr) return false;
320     const curDateStr = `${y}-${String(m + 1).padStart(2, '0')}-${String(d).padStart(2, '0')}`;
321     return curDateStr > sDateStr && curDateStr < eDateStr;
322   };
323 
324   const renderCalendar = (offset: number) => {
325     const baseDate = new Date(navDate.getFullYear(), navDate.getMonth() + offset, 1);
326     const y = baseDate.getFullYear();
327     const m = baseDate.getMonth();
328     const firstDay = new Date(y, m, 1).getDay();
329     const daysInMonth = new Date(y, m+1, 0).getDate();
330     
331     let days = [];
332     for(let i=0;i<firstDay;i++) days.push(null);
333     for(let i=1;i<=daysInMonth;i++) days.push(i);
334 
335     const monthName = baseDate.toLocaleString('default', { month: 'long', year: 'numeric' });
336 
337     return (
338       <div className="flex-1 w-52 bg-transparent">
339          <div className="font-semibold text-gray-800 text-sm mb-3 text-center">{monthName}</div>
340          <div className="grid grid-cols-7 gap-1 text-center text-xs text-gray-500 mb-2 font-medium">
341            {['Sun','Mon','Tue','Wed','Thu','Fri','Sat'].map(d=><div key={d}>{d}</div>)}
342          </div>
343          <div className="grid grid-cols-7 gap-y-1 gap-x-0 cursor-default">
344            {days.map((d, i) => {
345              if(!d) return <div key={i} />;
346              const isSel = isExactDate(y, m, d);
347              const inRange = isDateInRange(y, m, d);
348              
349              return (
350                <div 
351                  key={i} 
352                  onClick={() => handleDateClick(y, m, d)}
353                  className={`text-xs p-1.5 rounded-md text-center transition-colors cursor-pointer ${isSel ? 'bg-blue-600 text-white font-bold' : inRange ? 'bg-blue-50 text-blue-800' : 'text-gray-700 hover:bg-gray-100'}`}
354                >
355                  {d}
356                </div>
357              );
358            })}
359          </div>
360       </div>
361     )
362   }
363 
364   return (
365     <div ref={ref} className="relative">
366       <button onClick={() => { setTempFilt(dateFilt); setTempS(customS); setTempE(customE); setOpen(!open); }} className="flex items-center gap-2 bg-white border hover:bg-white/70 px-4.5 py-2 rounded-full border-black/10 shadow-sm cursor-pointer hover:bg-gray-50 transition-colors text-xs font-bold">
367         <Calendar size={15} className="text-gray-500" />
368         <span className="text-gray-800">{activeLabel}</span>
369         <ChevronDown size={13} className="text-gray-500 ml-0.5" />
370       </button>
371 
372       {open && (
373         <div className="absolute top-full left-0 mt-2 bg-white rounded-2xl shadow-xl border border-black/10 z-[9999] overflow-hidden flex flex-col md:flex-row w-max text-left">
374           {/* Left Sidebar */}
375           <div className="w-44 bg-gray-50 border-r border-black/5 py-4 px-2.5 flex flex-col gap-1 overflow-y-auto max-h-[350px]">
376              {OPTIONS.map(o => (
377                <label key={o.id} className="flex items-center gap-2.5 px-3.5 py-2 rounded-md hover:bg-gray-100 cursor-pointer">
378                  <div className={`w-3.5 h-3.5 rounded-full border-2 flex items-center justify-center bg-white ${tempFilt === o.id ? 'border-blue-500' : 'border-gray-300'}`}>
379                    {tempFilt === o.id && <div className="w-1.5 h-1.5 rounded-full bg-blue-500"></div>}
380                  </div>
381                  <span className="text-xs font-semibold text-gray-700">{o.label}</span>
382                  <input type="radio" className="hidden" name="dateFiltRadio" value={o.id} checked={tempFilt === o.id} onChange={() => setTempFilt(o.id)} />
383                </label>
384              ))}
385           </div>
386 
387           {/* Right Area */}
388           <div className="flex flex-col p-6 max-w-lg bg-transparent">
389              {/* Calendars */}
390              <div className="flex gap-4 relative justify-center">
391                <button onClick={() => setNavDate(new Date(navDate.getFullYear(), navDate.getMonth()-1, 1))} className="absolute -left-1 top-0 p-1 hover:bg-gray-100 rounded-full"><ChevronLeft size={16}/></button>
392                {renderCalendar(0)}
393                <button onClick={() => setNavDate(new Date(navDate.getFullYear(), navDate.getMonth()+1, 1))} className="absolute -right-1 top-0 p-1 hover:bg-gray-100 rounded-full"><ChevronRight size={16}/></button>
394              </div>
395 
396              {/* Footer Form */}
397              <div className="mt-4 pt-3 border-t border-gray-100 flex flex-col gap-3">
398                <div className="flex items-center justify-between gap-4">
399                  {tempFilt === "custom" ? (
400                    <div className="flex gap-1.5 items-center">
401                      <input type="date" value={tempS} onChange={(e)=>setTempS(e.target.value)} className="text-xs px-2 py-1 bg-white border border-gray-300 rounded outline-none focus:border-blue-500 font-medium"/>
402                      <span className="text-gray-400 font-semibold text-xs">-</span>
403                      <input type="date" value={tempE} onChange={(e)=>setTempE(e.target.value)} className="text-xs px-2 py-1 bg-white border border-gray-300 rounded outline-none focus:border-blue-500 font-medium"/>
404                    </div>
405                  ) : (
406                    <div className="font-semibold text-gray-800 text-sm">{OPTIONS.find(o=>o.id===tempFilt)?.label}</div>
407                  )}
408                </div>
409                
410                <div className="flex items-center gap-2 self-end">
411                  <button onClick={handleCancel} className="px-4 py-1.5 rounded-lg border border-gray-300 text-xs font-semibold text-gray-700 hover:bg-gray-50 transition-colors">Cancel</button>
412                  <button onClick={handleUpdate} className="px-4 py-1.5 rounded-lg bg-blue-600 border border-blue-600 text-xs font-semibold text-white hover:bg-blue-700 transition-colors">Update</button>
413                </div>
414              </div>
415           </div>

689 function getAggregatedDemographics(demographicsState: any, platformsList: any[]) {
690   const filledKeys = (platformsList || []).map((p: any) => {
691     const name = typeof p === 'string' ? p : p.name;
692     return { name, key: name.toLowerCase() };
693   }).filter(item => !!demographicsState[item.key]);
694 
695   if (filledKeys.length === 0) {
696     return null;
697   }
698 
699   const numPlatforms = filledKeys.length;
700   
701   let totalFemale = 0;
702   let totalMale = 0;
703   
704   const ageSums: { [range: string]: number } = {
705     "13-17": 0, "18-24": 0, "25-34": 0, "35-44": 0, "45+": 0
706   };
707 
708   const citySums: { [name: string]: number } = {};
709   const countrySums: { [name: string]: number } = {};
710   const deviceSums: { [name: string]: number } = {};
711   const interestSums: { [name: string]: number } = {};
712 
713   filledKeys.forEach(item => {
714     const data = demographicsState[item.key];
715     if (!data) return;
716     
717     // Gender
718     totalFemale += (data.gender?.female !== undefined ? data.gender.female : 50);
719     totalMale += (data.gender?.male !== undefined ? data.gender.male : 50);
720     
721     // Age
722     if (Array.isArray(data.age)) {
723       data.age.forEach((g: any) => {
724         if (g.range && g.value !== undefined) {
725           ageSums[g.range] = (ageSums[g.range] || 0) + g.value;
726         }
727       });
728     }
729 
730     // Cities
731     if (Array.isArray(data.cities)) {
732       data.cities.forEach((c: any) => {
733         if (c.name && c.percentage !== undefined) {
734           const normName = c.name.trim();
735           if (normName) {
736             citySums[normName] = (citySums[normName] || 0) + c.percentage;
737           }
738         }
739       });
740     }
741 
742     // Countries
743     if (Array.isArray(data.countries)) {
744       data.countries.forEach((c: any) => {
745         if (c.name && c.percentage !== undefined) {
746           const normName = c.name.trim();
747           if (normName) {
748             countrySums[normName] = (countrySums[normName] || 0) + c.percentage;
749           }
750         }
751       });
752     }
753 
754     // Devices
755     if (Array.isArray(data.devices)) {
756       data.devices.forEach((d: any) => {
757         if (d.name && d.percentage !== undefined) {
758           const normName = d.name.trim();
759           if (normName) {
760             deviceSums[normName] = (deviceSums[normName] || 0) + d.percentage;
761           }
762         }
763       });
764     }
765 
766     // Interests
767     if (Array.isArray(data.interests)) {
768       data.interests.forEach((n: any) => {
769         if (n.name && n.percentage !== undefined) {
770           const normName = n.name.trim();
771           if (normName) {
772             interestSums[normName] = (interestSums[normName] || 0) + n.percentage;
773           }
774         }
775       });
776     }
777   });
778 
779   const femaleAvg = Math.round(totalFemale / numPlatforms);
780   const maleAvg = 100 - femaleAvg;
781 
782   const ageList = Object.keys(ageSums).map(range => ({
783     range,
784     value: Math.round(ageSums[range] / numPlatforms)
785   }));
786 
787   const citiesList = Object.keys(citySums).map(name => ({
788     name,
789     percentage: Math.round(citySums[name] / numPlatforms)
790   })).sort((a, b) => b.percentage - a.percentage).slice(0, 5);
791 
792   const countriesList = Object.keys(countrySums).map(name => ({
793     name,
794     percentage: Math.round(countrySums[name] / numPlatforms)
795   })).sort((a, b) => b.percentage - a.percentage).slice(0, 5);
796 
797   const devicesList = Object.keys(deviceSums).map(name => ({
798     name,
799     percentage: Math.round(deviceSums[name] / numPlatforms)
800   })).sort((a, b) => b.percentage - a.percentage).slice(0, 3);
801 
802   const interestsList = Object.keys(interestSums).map(name => ({
803     name,
804     percentage: Math.round(interestSums[name] / numPlatforms)
805   })).sort((a, b) => b.percentage - a.percentage).slice(0, 5);
806 
807   return {
808     platform: (lang === "id" ? "Semua Platform" : "All Platforms"),
809     gender: { male: maleAvg, female: femaleAvg },
810     age: ageList,
811     cities: citiesList,
812     countries: countriesList,
813     devices: devicesList,
814     interests: interestsList
815   };
816 }
